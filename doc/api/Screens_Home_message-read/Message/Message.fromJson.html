factory Message.fromJson(dynamic json) {
  return Message.generate(
    json["id"] as int,
    json["sent"] as String,
    json["message"] as String,
    json["to"] as String,
    json["course"] as int,
    json["by"] as int,
    json["read_by"].cast<int>() as List<int>,
    json["from_username"] as String,
    json["is_professor"] as bool,
    json["course_name"] as String,
  );
}
